7/** Faciliates running rewrites over an [`EGraph`].
8
9One use for [`EGraph`]s is as the basis of a rewriting system.
10Since an egraph never "forgets" state when applying a [`Rewrite`], you
11can apply many rewrites many times quite efficiently.
12After the egraph is "full" (the rewrites can no longer find new
13equalities) or some other condition, the egraph compactly represents
14many, many equivalent expressions.
15At this point, the egraph is ready for extraction (see [`Extractor`])
16which can pick the represented expression that's best according to
17some cost function.
18
19This technique is called
20[equality saturation](https://www.cs.cornell.edu/~ross/publications/eqsat/)
21in general.
22However, there can be many challenges in implementing this "outer
23loop" of applying rewrites, mostly revolving around which rules to run
24and when to stop.
25
26[`Runner`] is `egg`'s provided equality saturation engine that has
27reasonable defaults and implements many useful things like saturation
28checking, egraph size limits, and customizable rule
29[scheduling](RewriteScheduler).
30Consider using [`Runner`] before rolling your own outer loop.
31
32Here are some of the things [`Runner`] does for you:
33
34- Saturation checking
35
36  [`Runner`] checks to see if any of the rules added anything
37  new to the [`EGraph`]. If none did, then it stops, returning
38  [`StopReason::Saturated`].
39
40- Iteration limits
41
42  You can set a upper limit of iterations to do in case the search
43  doesn't stop for some other reason. If this limit is hit, it stops with
44  [`StopReason::IterationLimit`].
45
46- [`EGraph`] size limit
47
48  You can set a upper limit on the number of enodes in the egraph.
49  If this limit is hit, it stops with
50  [`StopReason::NodeLimit`].
51
52- Time limit
53
54  You can set a time limit on the runner.
55  If this limit is hit, it stops with
56  [`StopReason::TimeLimit`].
57
58- Rule scheduling
59
60  Some rules enable themselves, blowing up the [`EGraph`] and
61  preventing other rewrites from running as many times.
62  To prevent this, you can provide your own [`RewriteScheduler`] to
63  govern when to run which rules.
64
65  [`BackoffScheduler`] is the default scheduler.
66
67[`Runner`] generates [`Iteration`]s that record some data about
68each iteration.
69You can add your own data to this by implementing the
70[`IterationData`] trait.
71[`Runner`] is generic over the [`IterationData`] that it will be in the
72[`Iteration`]s, but by default it uses `()`.
73
74
75# Example
76
77```
78use egg::{*, rewrite as rw};
79
80define_language! {
81    enum SimpleLanguage {
82        Num(i32),
83        "+" = Add([Id; 2]),
84        "*" = Mul([Id; 2]),
85        Symbol(Symbol),
86    }
87}
88
89let rules: &[Rewrite<SimpleLanguage, ()>] = &[
90    rw!("commute-add"; "(+ ?a ?b)" => "(+ ?b ?a)"),
91    rw!("commute-mul"; "(* ?a ?b)" => "(* ?b ?a)"),
92
93    rw!("add-0"; "(+ ?a 0)" => "?a"),
94    rw!("mul-0"; "(* ?a 0)" => "0"),
95    rw!("mul-1"; "(* ?a 1)" => "?a"),
96];
97
98pub struct MyIterData {
99    smallest_so_far: usize,
100}
101
102type MyRunner = Runner<SimpleLanguage, (), MyIterData>;
103
104impl IterationData<SimpleLanguage, ()> for MyIterData {
105    fn make(runner: &MyRunner) -> Self {
106        let root = runner.roots[0];
107        let mut extractor = Extractor::new(&runner.egraph, AstSize);
108        MyIterData {
109            smallest_so_far: extractor.find_best(root).0,
110        }
111    }
112}
113
114let start = "(+ 0 (* 1 foo))".parse().unwrap();
115// Runner is customizable in the builder pattern style.
116let runner = MyRunner::new(Default::default())
117    .with_iter_limit(10)
118    .with_node_limit(10_000)
119    .with_expr(&start)
120    .with_scheduler(SimpleScheduler)
121    .run(rules);
122
123// Now we can check our iteration data to make sure that the cost only
124// got better over time
125for its in runner.iterations.windows(2) {
126    assert!(its[0].data.smallest_so_far >= its[1].data.smallest_so_far);
127}
128
129println!(
130    "Stopped after {} iterations, reason: {:?}",
131    runner.iterations.len(),
132    runner.stop_reason
133);
134
135```
136*/
137pub struct Runner<L: Language, N: Analysis<L>, IterData = ()> {
138    /// The [`EGraph`] used.
139    pub egraph: EGraph<L, N>,
140    /// Data accumulated over each [`Iteration`].
141    pub iterations: Vec<Iteration<IterData>>,
142    /// The roots of expressions added by the
143    /// [`with_expr`](Runner::with_expr()) method, in insertion order.
144    pub roots: Vec<Id>,
145    /// Why the `Runner` stopped. This will be `None` if it hasn't
146    /// stopped yet.
147    pub stop_reason: Option<StopReason>,
148
149    /// The hooks added by the
150    /// [`with_hook`](Runner::with_hook()) method, in insertion order.
151    #[allow(clippy::type_complexity)]
152    pub hooks: Vec<Box<dyn FnMut(&mut Self) -> Result<(), String>>>,
153
154    limits: RunnerLimits,
155    scheduler: Box<dyn RewriteScheduler<L, N>>,
156}

158/// Describes the limits that would stop a [`Runner`].
159#[derive(Debug)]
160pub struct RunnerLimits {
161    iter_limit: usize,
162    node_limit: usize,
163    time_limit: Duration,
164    start_time: Option<Instant>,
165}
166
167impl RunnerLimits {
168    /// Check if the [`Runner`] should stop based on the limits.
169    pub fn check_limits<L, N>(&self, iteration: usize, egraph: &EGraph<L, N>) -> RunnerResult<()>
170    where
171        L: Language,
172        N: Analysis<L>,
173    {
174        let elapsed = self.start_time.unwrap().elapsed();
175        if elapsed > self.time_limit {
176            return Err(StopReason::TimeLimit(elapsed.as_secs_f64()));
177        }
178
179        let size = egraph.total_size();
180        if size > self.node_limit {
181            return Err(StopReason::NodeLimit(size));
182        }
183
184        if iteration >= self.iter_limit {
185            return Err(StopReason::IterationLimit(iteration));
186        }
187
188        Ok(())
189    }
190}

232/// Error returned by [`Runner`] when it stops.
233///
234#[derive(Debug, Clone)]
235#[cfg_attr(feature = "serde-1", derive(serde::Serialize))]
236pub enum StopReason {
237    /// The egraph saturated, i.e., there was an iteration where we
238    /// didn't learn anything new from applying the rules.
239    Saturated,
240    /// The iteration limit was hit. The data is the iteration limit.
241    IterationLimit(usize),
242    /// The enode limit was hit. The data is the enode limit.
243    NodeLimit(usize),
244    /// The time limit was hit. The data is the time limit in seconds.
245    TimeLimit(f64),
246    /// Some other reason to stop.
247    Other(String),
248}

429    /// Run this `Runner` until it stops.
430    /// After this, the field
431    /// [`stop_reason`](Runner::stop_reason) is guaranteed to be
432    /// set.
433    pub fn run<'a, R>(mut self, rules: R) -> Self
434    where
435        R: IntoIterator<Item = &'a Rewrite<L, N>>,
436        L: 'a,
437        N: 'a,
438    {
439        let rules: Vec<&Rewrite<L, N>> = rules.into_iter().collect();
440        check_rules(&rules);
441        self.egraph.rebuild();
442        loop {
443            let iter = self.run_one(&rules);
444            self.iterations.push(iter);
445            let stop_reason = self.iterations.last().unwrap().stop_reason.clone();
446            // we need to check_limits after the iteration is complete to check for iter_limit
447            if let Some(stop_reason) = stop_reason.or_else(|| self.check_limits().err()) {
448                info!("Stopping: {:?}", stop_reason);
449                self.stop_reason = Some(stop_reason);
450                break;
451            }
452        }
453
454        assert!(!self.iterations.is_empty());
455        assert!(self.stop_reason.is_some());
456        self
457    }

523    fn run_one(&mut self, rules: &[&Rewrite<L, N>]) -> Iteration<IterData> {
524        assert!(self.stop_reason.is_none());
525
526        info!("\nIteration {}", self.iterations.len());
527
528        self.try_start();
529        let mut result = self.check_limits();
530
531        let egraph_nodes = self.egraph.total_size();
532        let egraph_classes = self.egraph.number_of_classes();
533
534        let hook_time = Instant::now();
535        let mut hooks = std::mem::take(&mut self.hooks);
536        result = result.and_then(|_| {
537            hooks
538                .iter_mut()
539                .try_for_each(|hook| hook(self).map_err(StopReason::Other))
540        });
541        self.hooks = hooks;
542        let hook_time = hook_time.elapsed().as_secs_f64();
543
544        let egraph_nodes_after_hooks = self.egraph.total_size();
545        let egraph_classes_after_hooks = self.egraph.number_of_classes();
546
547        let i = self.iterations.len();
548        trace!("EGraph {:?}", self.egraph.dump());
549
550        let start_time = Instant::now();
551
552        let mut matches = Vec::new();
553        let mut applied = IndexMap::default();
554        result = result.and_then(|_| {
555            matches = self
556                .scheduler
557                .search_rewrites(i, &self.egraph, rules, &self.limits)?;
558            Ok(())
559            // rules.iter().try_for_each(|rw| {
560            //     let ms = self.scheduler.search_rewrite(i, &self.egraph, rw);
561            //     matches.push(ms);
562            //     self.check_limits()
563            // })
564        });
565
566        let search_time = start_time.elapsed().as_secs_f64();
567        info!("Search time: {}", search_time);
568
569        let apply_time = Instant::now();
570
571        result = result.and_then(|_| {
572            rules.iter().zip(matches).try_for_each(|(rw, ms)| {
573                let total_matches: usize = ms.iter().map(|m| m.substs.len()).sum();
574                debug!("Applying {} {} times", rw.name, total_matches);
575
576                let actually_matched = self.scheduler.apply_rewrite(i, &mut self.egraph, rw, ms);
577                if actually_matched > 0 {
578                    if let Some(count) = applied.get_mut(&rw.name) {
579                        *count += actually_matched;
580                    } else {
581                        applied.insert(rw.name.to_owned(), actually_matched);
582                    }
583                    debug!("Applied {} {} times", rw.name, actually_matched);
584                }
585                self.check_limits()
586            })
587        });
588
589        let apply_time = apply_time.elapsed().as_secs_f64();
590        info!("Apply time: {}", apply_time);
591
592        let rebuild_time = Instant::now();
593        let n_rebuilds = self.egraph.rebuild();
594        if self.egraph.are_explanations_enabled() {
595            debug_assert!(self.egraph.check_each_explain(rules));
596        }
597
598        let rebuild_time = rebuild_time.elapsed().as_secs_f64();
599        info!("Rebuild time: {}", rebuild_time);
600        info!(
601            "Size: n={}, e={}",
602            self.egraph.total_size(),
603            self.egraph.number_of_classes()
604        );
605
606        let can_be_saturated = applied.is_empty()
607            && self.scheduler.can_stop(i)
608            // now make sure the hooks didn't do anything
609            && (egraph_nodes == egraph_nodes_after_hooks)
610            && (egraph_classes == egraph_classes_after_hooks)
611            // now make sure that conditional rules (which might add
612            // nodes without applying) didn't do anything
613            && (egraph_nodes == self.egraph.total_size())
614            && (egraph_classes == self.egraph.number_of_classes());
615
616        if can_be_saturated {
617            result = result.and(Err(StopReason::Saturated))
618        }
619
620        Iteration {
621            applied,
622            egraph_nodes,
623            egraph_classes,
624            hook_time,
625            search_time,
626            apply_time,
627            rebuild_time,
628            n_rebuilds,
629            data: IterData::make(self),
630            total_time: start_time.elapsed().as_secs_f64(),
631            stop_reason: result.err(),
632        }
633    }
634
635    fn try_start(&mut self) {
636        self.limits.start_time.get_or_insert_with(Instant::now);
637    }
638
639    fn check_limits(&self) -> RunnerResult<()> {
640        self.limits
641            .check_limits(self.iterations.len(), &self.egraph)
642    }

645fn check_rules<L, N>(rules: &[&Rewrite<L, N>]) {
646    let mut name_counts = IndexMap::default();
647    for rw in rules {
648        *name_counts.entry(rw.name).or_default() += 1
649    }
650
651    name_counts.retain(|_, count: &mut usize| *count > 1);
652    if !name_counts.is_empty() {
653        eprintln!("WARNING: Duplicated rule names may affect rule reporting and scheduling.");
654        log::warn!("Duplicated rule names may affect rule reporting and scheduling.");
655        for (name, &count) in name_counts.iter() {
656            assert!(count > 1);
657            eprintln!("Rule '{}' appears {} times", name, count);
658            log::warn!("Rule '{}' appears {} times", name, count);
659        }
660    }
661}

663/** A way to customize how a [`Runner`] runs [`Rewrite`]s.
664
665This gives you a way to prevent certain [`Rewrite`]s from exploding
666the [`EGraph`] and dominating how much time is spent while running the
667[`Runner`].
668
669*/
670#[allow(unused_variables)]
671pub trait RewriteScheduler<L, N>
672where
673    L: Language,
674    N: Analysis<L>,
675{
676    /// Whether or not the [`Runner`] is allowed
677    /// to say it has saturated.
678    ///
679    /// This is only called when the runner is otherwise saturated.
680    /// Default implementation just returns `true`.
681    fn can_stop(&mut self, iteration: usize) -> bool {
682        true
683    }
684
685    /// A hook allowing you to customize rewrite searching behavior.
686    /// Useful to implement rule management.
687    ///
688    /// Default implementation just calls
689    /// [`Rewrite::search`](Rewrite::search()).
690    fn search_rewrite<'a>(
691        &mut self,
692        iteration: usize,
693        egraph: &EGraph<L, N>,
694        rewrite: &'a Rewrite<L, N>,
695    ) -> Vec<SearchMatches<'a, L>> {
696        rewrite.search(egraph)
697    }
698
699    /// A hook allowing you to customize rewrite searching behavior
700    /// across rewrites.
701    ///
702    /// Default implementation calls
703    /// [`Self::search_rewrite`] for each rewrite,
704    /// and checks [`RunnerLimits::check_limits`] after each.
705    ///
706    /// Returning an error will stop the runner.
707    ///
708    /// You might use this to implement parallel rule application:
709    /// ```
710    /// # use egg::*;
711    /// pub struct ParallelRewriteScheduler;
712    /// impl RewriteScheduler<SymbolLang, ()> for ParallelRewriteScheduler {
713    ///     fn search_rewrites<'a>(
714    ///         &mut self,
715    ///         iteration: usize,
716    ///         egraph: &EGraph<SymbolLang, ()>,
717    ///         rewrites: &[&'a Rewrite<SymbolLang, ()>],
718    ///         _limits: &RunnerLimits,
719    ///     ) -> RunnerResult<Vec<Vec<SearchMatches<'a, SymbolLang>>>> {
720    ///         // this implementation just ignores the limits
721    ///         // fake `par_map` to enforce Send + Sync, in real life use rayon
722    ///         fn par_map<T, F, T2>(slice: &[T], f: F) -> Vec<T2>
723    ///         where
724    ///             T: Send + Sync,
725    ///             F: Fn(&T) -> T2 + Send + Sync,
726    ///             T2: Send + Sync,
727    ///         {
728    ///             slice.iter().map(f).collect()
729    ///         }
730    ///         Ok(par_map(rewrites, |rw| rw.search(egraph)))
731    ///     }
732    /// }
733    /// ```
734    fn search_rewrites<'a>(
735        &mut self,
736        iteration: usize,
737        egraph: &EGraph<L, N>,
738        rewrites: &[&'a Rewrite<L, N>],
739        limits: &RunnerLimits,
740    ) -> RunnerResult<Vec<Vec<SearchMatches<'a, L>>>> {
741        let mut matches = Vec::new();
742        for rw in rewrites {
743            let ms = self.search_rewrite(iteration, egraph, rw);
744            matches.push(ms);
745            limits.check_limits(iteration, egraph)?;
746        }
747        Ok(matches)
748    }
749
750    /// A hook allowing you to customize rewrite application behavior.
751    /// Useful to implement rule management.
752    ///
753    /// Default implementation just calls
754    /// [`Rewrite::apply`](Rewrite::apply())
755    /// and returns number of new applications.
756    fn apply_rewrite(
757        &mut self,
758        iteration: usize,
759        egraph: &mut EGraph<L, N>,
760        rewrite: &Rewrite<L, N>,
761        matches: Vec<SearchMatches<L>>,
762    ) -> usize {
763        rewrite.apply(egraph, &matches).len()
764    }
765}

769///
770/// Using this is basically turning off rule scheduling.
771/// It uses the default implementation for all [`RewriteScheduler`]
772/// methods.
773///
774/// This is not the default scheduler; choose it with the
775/// [`with_scheduler`](Runner::with_scheduler())
776/// method.
777///
778#[derive(Debug)]
779pub struct SimpleScheduler;

781impl<L, N> RewriteScheduler<L, N> for SimpleScheduler
782where
783    L: Language,
784    N: Analysis<L>,

788/// A [`RewriteScheduler`] that implements exponentional rule backoff.
789///
790/// For each rewrite, there exists a configurable initial match limit.
791/// If a rewrite search yield more than this limit, then we ban this
792/// rule for number of iterations, double its limit, and double the time
793/// it will be banned next time.
794///
795/// This seems effective at preventing explosive rules like
796/// associativity from taking an unfair amount of resources.
797///
798/// [`BackoffScheduler`] is configurable in the builder-pattern style.
799///
800#[derive(Debug)]
801pub struct BackoffScheduler {
802    default_match_limit: usize,
803    default_ban_length: usize,
804    stats: IndexMap<Symbol, RuleStats>,
805}
806
807#[derive(Debug)]
808struct RuleStats {
809    times_applied: usize,
810    banned_until: usize,
811    times_banned: usize,
812    match_limit: usize,
813    ban_length: usize,
814}
815
816impl BackoffScheduler {
817    /// Set the initial match limit after which a rule will be banned.
818    /// Default: 1,000
819    pub fn with_initial_match_limit(mut self, limit: usize) -> Self {
820        self.default_match_limit = limit;
821        self
822    }
823
824    /// Set the initial ban length.
825    /// Default: 5 iterations
826    pub fn with_ban_length(mut self, ban_length: usize) -> Self {
827        self.default_ban_length = ban_length;
828        self
829    }
830
831    fn rule_stats(&mut self, name: Symbol) -> &mut RuleStats {
832        if self.stats.contains_key(&name) {
833            &mut self.stats[&name]
834        } else {
835            self.stats.entry(name).or_insert(RuleStats {
836                times_applied: 0,
837                banned_until: 0,
838                times_banned: 0,
839                match_limit: self.default_match_limit,
840                ban_length: self.default_ban_length,
841            })
842        }
843    }
844
845    /// Never ban a particular rule.
846    pub fn do_not_ban(mut self, name: impl Into<Symbol>) -> Self {
847        self.rule_stats(name.into()).match_limit = usize::MAX;
848        self
849    }
850
851    /// Set the initial match limit for a rule.
852    pub fn rule_match_limit(mut self, name: impl Into<Symbol>, limit: usize) -> Self {
853        self.rule_stats(name.into()).match_limit = limit;
854        self
855    }
856
857    /// Set the initial ban length for a rule.
858    pub fn rule_ban_length(mut self, name: impl Into<Symbol>, length: usize) -> Self {
859        self.rule_stats(name.into()).ban_length = length;
860        self
861    }
862}
863
864impl Default for BackoffScheduler {
865    fn default() -> Self {
866        Self {
867            stats: Default::default(),
868            default_match_limit: 1_000,
869            default_ban_length: 5,
870        }
871    }
872}
873
874impl<L, N> RewriteScheduler<L, N> for BackoffScheduler
875where
876    L: Language,
877    N: Analysis<L>,
878{
879    fn can_stop(&mut self, iteration: usize) -> bool {
880        let n_stats = self.stats.len();
881
882        let mut banned: Vec<_> = self
883            .stats
884            .iter_mut()
885            .filter(|(_, s)| s.banned_until > iteration)
886            .collect();
887
888        if banned.is_empty() {
889            true
890        } else {
891            let min_ban = banned
892                .iter()
893                .map(|(_, s)| s.banned_until)
894                .min()
895                .expect("banned cannot be empty here");
896
897            assert!(min_ban >= iteration);
898            let delta = min_ban - iteration;
899
900            let mut unbanned = vec![];
901            for (name, s) in &mut banned {
902                s.banned_until -= delta;
903                if s.banned_until == iteration {
904                    unbanned.push(name.as_str());
905                }
906            }
907
908            assert!(!unbanned.is_empty());
909            info!(
910                "Banned {}/{}, fast-forwarded by {} to unban {}",
911                banned.len(),
912                n_stats,
913                delta,
914                unbanned.join(", "),
915            );
916
917            false
918        }
919    }
920
921    fn search_rewrite<'a>(
922        &mut self,
923        iteration: usize,
924        egraph: &EGraph<L, N>,
925        rewrite: &'a Rewrite<L, N>,
926    ) -> Vec<SearchMatches<'a, L>> {
927        let stats = self.rule_stats(rewrite.name);
928
929        if iteration < stats.banned_until {
930            debug!(
931                "Skipping {} ({}-{}), banned until {}...",
932                rewrite.name, stats.times_applied, stats.times_banned, stats.banned_until,
933            );
934            return vec![];
935        }
936
937        let threshold = stats
938            .match_limit
939            .checked_shl(stats.times_banned as u32)
940            .unwrap();
941        let matches = rewrite.search_with_limit(egraph, threshold.saturating_add(1));
942        let total_len: usize = matches.iter().map(|m| m.substs.len()).sum();
943        if total_len > threshold {
944            let ban_length = stats.ban_length << stats.times_banned;
945            stats.times_banned += 1;
946            stats.banned_until = iteration + ban_length;
947            info!(
948                "Banning {} ({}-{}) for {} iters: {} < {}",
949                rewrite.name,
950                stats.times_applied,
951                stats.times_banned,
952                ban_length,
953                threshold,
954                total_len,
955            );
956            vec![]
957        } else {
958            stats.times_applied += 1;
959            matches
960        }
961    }
962}
