2use std::iter::ExactSizeIterator;

4use crate::*;

6/// An equivalence class of enodes.
7#[non_exhaustive]
8#[derive(Debug, Clone)]
9#[cfg_attr(feature = "serde-1", derive(serde::Serialize, serde::Deserialize))]
10pub struct EClass<L, D> {
11    /// This eclass's id.
12    pub id: Id,
13    /// The equivalent enodes in this equivalence class.
14    pub nodes: Vec<L>,
15    /// The analysis data associated with this eclass.
16    ///
17    /// Modifying this field will _not_ cause changes to propagate through the e-graph.
18    /// Prefer [`EGraph::set_analysis_data`] instead.
19    pub data: D,
20    /// The original Ids of parent enodes.
21    pub(crate) parents: Vec<Id>,
22}
23
24impl<L, D> EClass<L, D> {
25    /// Returns `true` if the `eclass` is empty.
26    pub fn is_empty(&self) -> bool {
27        self.nodes.is_empty()
28    }
29
30    /// Returns the number of enodes in this eclass.
31    pub fn len(&self) -> usize {
32        self.nodes.len()
33    }
34
35    /// Iterates over the enodes in this eclass.
36    pub fn iter(&self) -> impl ExactSizeIterator<Item = &L> {
37        self.nodes.iter()
38    }
39
40    /// Iterates over the non-canonical ids of parent enodes of this eclass.
41    pub fn parents(&self) -> impl ExactSizeIterator<Item = Id> + '_ {
42        self.parents.iter().copied()
43    }
44}
45
46impl<L: Language, D> EClass<L, D> {
47    /// Iterates over the childless enodes in this eclass.
48    pub fn leaves(&self) -> impl Iterator<Item = &L> {
49        self.nodes.iter().filter(|&n| n.is_leaf())
50    }
51
52    /// Asserts that the childless enodes in this eclass are unique.
53    pub fn assert_unique_leaves(&self)
54    where
55        L: Language,
56    {
57        let mut leaves = self.leaves();
58        if let Some(first) = leaves.next() {
59            assert!(
60                leaves.all(|l| l == first),
61                "Different leaves in eclass {}: {:?}",
62                self.id,
63                self.leaves().collect::<crate::util::HashSet<_>>()
64            );
65        }
66    }
67
68    /// Run some function on each matching e-node in this class.
69    pub fn for_each_matching_node<Err>(
70        &self,
71        node: &L,
72        mut f: impl FnMut(&L) -> Result<(), Err>,
73    ) -> Result<(), Err>
74    where
75        L: Language,
76    {
77        if self.nodes.len() < 50 {
78            self.nodes
79                .iter()
80                .filter(|n| node.matches(n))
81                .try_for_each(f)
82        } else {
83            debug_assert!(node.all(|id| id == Id::from(0)));
84            debug_assert!(self.nodes.windows(2).all(|w| w[0] < w[1]));
85            let mut start = self.nodes.binary_search(node).unwrap_or_else(|i| i);
86            let discrim = node.discriminant();
87            while start > 0 {
88                if self.nodes[start - 1].discriminant() == discrim {
89                    start -= 1;
90                } else {
91                    break;
92                }
93            }
94            let mut matching = self.nodes[start..]
95                .iter()
96                .take_while(|&n| n.discriminant() == discrim)
97                .filter(|n| node.matches(n));
98            debug_assert_eq!(
99                matching.clone().count(),
100                self.nodes.iter().filter(|n| node.matches(n)).count(),
101                "matching node {:?}\nstart={}\n{:?} != {:?}\nnodes: {:?}",
102                node,
103                start,
104                matching.clone().collect::<HashSet<_>>(),
105                self.nodes
106                    .iter()
107                    .filter(|n| node.matches(n))
108                    .collect::<HashSet<_>>(),
109                self.nodes
110            );
111            matching.try_for_each(&mut f)
112        }
113    }
114}
