22#[derive(Debug, Clone, PartialEq, Eq)]
23enum Instruction<L> {
24    Bind { node: L, i: Reg, out: Reg },
25    Compare { i: Reg, j: Reg },
26    Lookup { term: Vec<ENodeOrReg<L>>, i: Reg },
27    Scan { out: Reg },
28}

30#[derive(Debug, Clone, PartialEq, Eq)]
31enum ENodeOrReg<L> {
32    ENode(L),
33    Reg(Reg),
34}

36impl Machine {
37    #[inline(always)]
38    fn reg(&self, reg: Reg) -> Id {
39        self.reg[reg.0 as usize]
40    }
41
42    fn run<L, N>(
43        &mut self,
44        egraph: &EGraph<L, N>,
45        instructions: &[Instruction<L>],
46        subst: &Subst,
47        yield_fn: &mut impl FnMut(&Self, &Subst) -> Result,
48    ) -> Result
49    where
50        L: Language,
51        N: Analysis<L>,
52    {
53        let mut instructions = instructions.iter();
54        while let Some(instruction) = instructions.next() {
55            match instruction {
56                Instruction::Bind { i, out, node } => {
57                    let remaining_instructions = instructions.as_slice();
58                    let eclass = &egraph[self.reg(*i)];
59                    return eclass.for_each_matching_node(node, |matched| {
60                        self.reg.truncate(out.0 as usize);
61                        matched.for_each(|id| self.reg.push(id));
62                        self.run(egraph, remaining_instructions, subst, yield_fn)
63                    });
64                }
65                Instruction::Scan { out } => {
66                    let remaining_instructions = instructions.as_slice();
67                    for class in egraph.classes() {
68                        self.reg.truncate(out.0 as usize);
69                        self.reg.push(class.id);
70                        self.run(egraph, remaining_instructions, subst, yield_fn)?
71                    }
72                    return Ok(());
73                }
74                Instruction::Compare { i, j } => {
75                    if egraph.find(self.reg(*i)) != egraph.find(self.reg(*j)) {
76                        return Ok(());
77                    }
78                }
79                Instruction::Lookup { term, i } => {
80                    self.lookup.clear();
81                    for node in term {
82                        match node {
83                            ENodeOrReg::ENode(node) => {
84                                let look = |i| self.lookup[usize::from(i)];
85                                match egraph.lookup(node.clone().map_children(look)) {
86                                    Some(id) => self.lookup.push(id),
87                                    None => return Ok(()),
88                                }
89                            }
90                            ENodeOrReg::Reg(r) => {
91                                self.lookup.push(egraph.find(self.reg(*r)));
92                            }
93                        }
94                    }
95
96                    let id = egraph.find(self.reg(*i));
97                    if self.lookup.last().copied() != Some(id) {
98                        return Ok(());
99                    }
100                }
101            }
102        }
103
104        yield_fn(self, subst)
105    }
106}

108struct Compiler<L> {
109    v2r: IndexMap<Var, Reg>,
110    free_vars: Vec<HashSet<Var>>,
111    subtree_size: Vec<usize>,
112    todo_nodes: HashMap<(Id, Reg), L>,
113    instructions: Vec<Instruction<L>>,
114    next_reg: Reg,
115}
116
117impl<L: Language> Compiler<L> {
118    fn new() -> Self {
119        Self {
120            free_vars: Default::default(),
121            subtree_size: Default::default(),
122            v2r: Default::default(),
123            todo_nodes: Default::default(),
124            instructions: Default::default(),
125            next_reg: Reg(0),
126        }
127    }
128
129    fn add_todo(&mut self, pattern: &PatternAst<L>, id: Id, reg: Reg) {
130        match &pattern[id] {
131            ENodeOrVar::Var(v) => {
132                if let Some(&j) = self.v2r.get(v) {
133                    self.instructions.push(Instruction::Compare { i: reg, j })
134                } else {
135                    self.v2r.insert(*v, reg);
136                }
137            }
138            ENodeOrVar::ENode(pat) => {
139                self.todo_nodes.insert((id, reg), pat.clone());
140            }
141        }
142    }
143
144    fn load_pattern(&mut self, pattern: &PatternAst<L>) {
145        let len = pattern.len();
146        self.free_vars = Vec::with_capacity(len);
147        self.subtree_size = Vec::with_capacity(len);
148
149        for node in pattern {
150            let mut free = HashSet::default();
151            let mut size = 0;
152            match node {
153                ENodeOrVar::ENode(n) => {
154                    size = 1;
155                    for &child in n.children() {
156                        free.extend(&self.free_vars[usize::from(child)]);
157                        size += self.subtree_size[usize::from(child)];
158                    }
159                }
160                ENodeOrVar::Var(v) => {
161                    free.insert(*v);
162                }
163            }
164            self.free_vars.push(free);
165            self.subtree_size.push(size);
166        }
167    }
168
169    fn next(&mut self) -> Option<((Id, Reg), L)> {
170        // we take the max todo according to this key
171        // - prefer grounded
172        // - prefer more free variables
173        // - prefer smaller term
174        let key = |(id, _): &&(Id, Reg)| {
175            let i = usize::from(*id);
176            let n_bound = self.free_vars[i]
177                .iter()
178                .filter(|v| self.v2r.contains_key(*v))
179                .count();
180            let n_free = self.free_vars[i].len() - n_bound;
181            let size = self.subtree_size[i] as isize;
182            (n_free == 0, n_free, -size)
183        };
184
185        self.todo_nodes
186            .keys()
187            .max_by_key(key)
188            .copied()
189            .map(|k| (k, self.todo_nodes.remove(&k).unwrap()))
190    }
191
192    /// check to see if this e-node corresponds to a term that is grounded by
193    /// the variables bound at this point
194    fn is_ground_now(&self, id: Id) -> bool {
195        self.free_vars[usize::from(id)]
196            .iter()
197            .all(|v| self.v2r.contains_key(v))
198    }
199
200    fn compile(&mut self, patternbinder: Option<Var>, pattern: &PatternAst<L>) {
201        self.load_pattern(pattern);
202        let root = pattern.root();
203
204        let mut next_out = self.next_reg;
205
206        // Check if patternbinder already bound in v2r
207        // Behavior common to creating a new pattern
208        let add_new_pattern = |comp: &mut Compiler<L>| {
209            if !comp.instructions.is_empty() {
210                // After first pattern needs scan
211                comp.instructions
212                    .push(Instruction::Scan { out: comp.next_reg });
213            }
214            comp.add_todo(pattern, root, comp.next_reg);
215        };
216
217        if let Some(v) = patternbinder {
218            if let Some(&i) = self.v2r.get(&v) {
219                // patternbinder already bound
220                self.add_todo(pattern, root, i);
221            } else {
222                // patternbinder is new variable
223                next_out.0 += 1;
224                add_new_pattern(self);
225                self.v2r.insert(v, self.next_reg); //add to known variables.
226            }
227        } else {
228            // No pattern binder
229            next_out.0 += 1;
230            add_new_pattern(self);
231        }
232
233        while let Some(((id, reg), node)) = self.next() {
234            if self.is_ground_now(id) && !node.is_leaf() {
235                let extracted = pattern.extract(id);
236                self.instructions.push(Instruction::Lookup {
237                    i: reg,
238                    term: extracted
239                        .iter()
240                        .map(|n| match n {
241                            ENodeOrVar::ENode(n) => ENodeOrReg::ENode(n.clone()),
242                            ENodeOrVar::Var(v) => ENodeOrReg::Reg(self.v2r[v]),
243                        })
244                        .collect(),
245                });
246            } else {
247                let out = next_out;
248                next_out.0 += node.len() as u32;
249
250                // zero out the children so Bind can use it to sort
251                let op = node.clone().map_children(|_| Id::from(0));
252                self.instructions.push(Instruction::Bind {
253                    i: reg,
254                    node: op,
255                    out,
256                });
257
258                for (i, &child) in node.children().iter().enumerate() {
259                    self.add_todo(pattern, child, Reg(out.0 + i as u32));
260                }
261            }
262        }
263        self.next_reg = next_out;
264    }
265
266    fn extract(self) -> Program<L> {
267        let mut subst = Subst::default();
268        for (v, r) in self.v2r {
269            subst.insert(v, Id::from(r.0 as usize));
270        }
271        Program {
272            instructions: self.instructions,
273            subst,
274        }
275    }
276}

278impl<L: Language> Program<L> {
279    pub(crate) fn compile_from_pat(pattern: &PatternAst<L>) -> Self {
280        let mut compiler = Compiler::new();
281        compiler.compile(None, pattern);
282        let program = compiler.extract();
283        log::debug!("Compiled {:?} to {:?}", pattern.as_ref(), program);
284        program
285    }
286
287    pub(crate) fn compile_from_multi_pat(patterns: &[(Var, PatternAst<L>)]) -> Self {
288        let mut compiler = Compiler::new();
289        for (var, pattern) in patterns {
290            compiler.compile(Some(*var), pattern);
291        }
292        compiler.extract()
293    }
294
295    pub fn run_with_limit<A>(
296        &self,
297        egraph: &EGraph<L, A>,
298        eclass: Id,
299        mut limit: usize,
300    ) -> Vec<Subst>
301    where
302        A: Analysis<L>,
303    {
304        assert!(egraph.clean, "Tried to search a dirty e-graph!");
305
306        if limit == 0 {
307            return vec![];
308        }
309
310        let mut machine = Machine::default();
311        assert_eq!(machine.reg.len(), 0);
312        machine.reg.push(eclass);
313
314        let mut matches = Vec::new();
315        machine
316            .run(
317                egraph,
318                &self.instructions,
319                &self.subst,
320                &mut |machine, subst| {
321                    if !egraph.analysis.allow_ematching_cycles() {
322                        if let Some((first, rest)) = machine.reg.split_first() {
323                            if rest.contains(first) {
324                                return Ok(());
325                            }
326                        }
327                    }
328
329                    let subst_vec = subst
330                        .vec
331                        .iter()
332                        // HACK we are reusing Ids here, this is bad
333                        .map(|(v, reg_id)| (*v, machine.reg(Reg(usize::from(*reg_id) as u32))))
334                        .collect();
335                    matches.push(Subst { vec: subst_vec });
336                    limit -= 1;
337                    if limit != 0 { Ok(()) } else { Err(()) }
338                },
339            )
340            .unwrap_or_default();
341
342        log::trace!("Ran program, found {:?}", matches);
343        matches
344    }
