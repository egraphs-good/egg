12/// An interned string.
13///
14/// This is provided by the [`symbol_table`](https://crates.io/crates/symbol_table) crate.
15///
16/// Internally, `egg` frequently compares [`Var`]s and elements of
17/// [`Language`]s. To keep comparisons fast, `egg` provides [`Symbol`] a simple
18/// wrapper providing interned strings.
19///
20/// You may wish to use [`Symbol`] in your own [`Language`]s to increase
21/// performance and keep enode sizes down (a [`Symbol`] is only 4 bytes,
22/// compared to 24 for a `String`.)
23///
24/// A [`Symbol`] is simply a wrapper around an integer.
25/// When creating a [`Symbol`] from a string, `egg` looks up it up in a global
26/// table, returning the index (inserting it if not found).
27/// That integer is used to cheaply implement
28/// `Copy`, `Clone`, `PartialEq`, `Eq`, `PartialOrd`, `Ord`, and `Hash`.
29///
30/// The internal symbol cache leaks the strings, which should be
31/// fine if you only put in things like variable names and identifiers.
32///
33/// # Example
34/// ```rust
35/// use egg::Symbol;
36///
37/// assert_eq!(Symbol::from("foo"), Symbol::from("foo"));
38/// assert_eq!(Symbol::from("foo"), "foo".parse().unwrap());
39///
40/// assert_ne!(Symbol::from("foo"), Symbol::from("bar"));
41/// ```
42///
43pub use symbol_table::GlobalSymbol as Symbol;

81pub(crate) fn pretty_print(
82    buf: &mut String,
83    sexp: &Sexp,
84    width: usize,
85    level: usize,
86) -> std::fmt::Result {
87    use std::fmt::Write;
88    if let Sexp::List(list) = sexp {
89        let indent = sexp.to_string().len() > width;
90        write!(buf, "(")?;
91
92        for (i, val) in list.iter().enumerate() {
93            if indent && i > 0 {
94                writeln!(buf)?;
95                for _ in 0..level {
96                    write!(buf, "  ")?;
97                }
98            }
99            pretty_print(buf, val, width, level + 1)?;
100            if !indent && i < list.len() - 1 {
101                write!(buf, " ")?;
102            }
103        }
104
105        write!(buf, ")")?;
106        Ok(())
107    } else {
108        // I don't care about quotes
109        write!(buf, "{}", sexp.to_string().trim_matches('"'))
110    }
111}

122/** A data structure to maintain a queue of unique elements.
123
124Notably, insert/pop operations have O(1) expected amortized runtime complexity.
125*/
126#[derive(Clone)]
127#[cfg_attr(feature = "serde-1", derive(Serialize, Deserialize))]
128pub(crate) struct UniqueQueue<T>
129where
130    T: Eq + std::hash::Hash + Clone,
131{
132    set: hashbrown::HashSet<T>,
133    queue: std::collections::VecDeque<T>,
134}
135
136impl<T> Default for UniqueQueue<T>
137where
138    T: Eq + std::hash::Hash + Clone,
139{
140    fn default() -> Self {
141        UniqueQueue {
142            set: hashbrown::HashSet::default(),
143            queue: std::collections::VecDeque::new(),
144        }
145    }
146}
147
148impl<T> UniqueQueue<T>
149where
150    T: Eq + std::hash::Hash + Clone,
151{
152    pub fn insert(&mut self, t: T) {
153        if self.set.insert(t.clone()) {
154            self.queue.push_back(t);
155        }
156    }
157
158    pub fn extend<I>(&mut self, iter: I)
159    where
160        I: IntoIterator<Item = T>,
161    {
162        for t in iter.into_iter() {
163            self.insert(t);
164        }
165    }
166
167    pub fn pop(&mut self) -> Option<T> {
168        let res = self.queue.pop_front();
169        res.as_ref().map(|t| self.set.remove(t));
170        res
171    }
172
173    pub fn is_empty(&self) -> bool {
174        let r = self.queue.is_empty();
175        debug_assert_eq!(r, self.set.is_empty());
176        r
177    }
178}
179
180impl<T> IntoIterator for UniqueQueue<T>
181where
182    T: Eq + std::hash::Hash + Clone,
183{
184    type Item = T;
185
186    type IntoIter = <std::collections::VecDeque<T> as IntoIterator>::IntoIter;
187
188    fn into_iter(self) -> Self::IntoIter {
189        self.queue.into_iter()
190    }
191}
192
193impl<A> FromIterator<A> for UniqueQueue<A>
194where
195    A: Eq + std::hash::Hash + Clone,
196{
197    fn from_iter<T: IntoIterator<Item = A>>(iter: T) -> Self {
198        let mut queue = UniqueQueue::default();
199        for t in iter {
200            queue.insert(t);
201        }
202        queue
203    }
