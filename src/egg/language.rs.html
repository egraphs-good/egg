17/// Trait that defines a Language whose terms will be in the [`EGraph`].
18///
19/// Check out the [`define_language!`] macro for an easy way to create
20/// a [`Language`].
21///
22/// If you want to pretty-print expressions, you should implement [`Display`] to
23/// display the language node's operator. For example, a language node
24/// `Add([Id; 2])` might be displayed as "+".
25///
26/// To parse expressions from strings you should also implement [`FromOp`].
27///
28/// The [`define_language!`] macro automatically implements both [`Display`] and
29/// [`FromOp`].
30///
31/// See [`SymbolLang`] for quick-and-dirty use cases.
32#[allow(clippy::len_without_is_empty)]
33pub trait Language: Debug + Clone + Eq + Ord + Hash {
34    /// Type representing the cases of this language.
35    ///
36    /// Used for short-circuiting the search for equivalent nodes.
37    type Discriminant: Debug + Clone + Eq + Hash;
38
39    /// Return the `Discriminant` of this node.
40    #[allow(enum_intrinsics_non_enums)]
41    fn discriminant(&self) -> Self::Discriminant;
42
43    /// Returns true if this enode matches another enode.
44    /// This should only consider the operator and the arity,
45    /// not the children `Id`s.
46    fn matches(&self, other: &Self) -> bool;
47
48    /// Returns the children of this e-node.
49    fn children(&self) -> &[Id];
50
51    /// Returns a mutable slice of the children of this e-node.
52    fn children_mut(&mut self) -> &mut [Id];
53
54    /// Runs a given function on each child `Id`.
55    fn for_each<F: FnMut(Id)>(&self, f: F) {
56        self.children().iter().copied().for_each(f)
57    }
58
59    /// Runs a given function on each child `Id`, allowing mutation of that `Id`.
60    fn for_each_mut<F: FnMut(&mut Id)>(&mut self, f: F) {
61        self.children_mut().iter_mut().for_each(f)
62    }
63
64    /// Runs a falliable function on each child, stopping if the function returns
65    /// an error.
66    fn try_for_each<E, F>(&self, mut f: F) -> Result<(), E>
67    where
68        F: FnMut(Id) -> Result<(), E>,
69    {
70        self.fold(Ok(()), |res, id| res.and_then(|_| f(id)))
71    }
72
73    /// Returns the number of the children this enode has.
74    ///
75    /// The default implementation uses `fold` to accumulate the number of
76    /// children.
77    fn len(&self) -> usize {
78        self.fold(0, |len, _| len + 1)
79    }
80
81    /// Returns true if this enode has no children.
82    fn is_leaf(&self) -> bool {
83        self.all(|_| false)
84    }
85
86    /// Runs a given function to replace the children.
87    fn update_children<F: FnMut(Id) -> Id>(&mut self, mut f: F) {
88        self.for_each_mut(|id| *id = f(*id))
89    }
90
91    /// Creates a new enode with children determined by the given function.
92    fn map_children<F: FnMut(Id) -> Id>(mut self, f: F) -> Self {
93        self.update_children(f);
94        self
95    }
96
97    /// Folds over the children, given an initial accumulator.
98    fn fold<F, T>(&self, init: T, mut f: F) -> T
99    where
100        F: FnMut(T, Id) -> T,
101    {
102        let mut acc = init;
103        for id in self.children().iter().copied() {
104            acc = f(acc, id)
105        }
106        acc
107    }
108
109    /// Returns true if the predicate is true on all children.
110    /// Does not short circuit.
111    fn all<F: FnMut(Id) -> bool>(&self, mut f: F) -> bool {
112        self.fold(true, |acc, id| acc && f(id))
113    }
114
115    /// Returns true if the predicate is true on any children.
116    /// Does not short circuit.
117    fn any<F: FnMut(Id) -> bool>(&self, mut f: F) -> bool {
118        self.fold(false, |acc, id| acc || f(id))
119    }
120
121    /// Make a [`RecExpr`] by mapping this enodes children to other [`RecExpr`]s.
122    ///
123    /// This can be used to join together different expression with a new node.
124    ///
125    /// # Example
126    /// ```
127    /// # use egg::*;
128    /// let a_plus_2: RecExpr<SymbolLang> = "(+ a 2)".parse().unwrap();
129    /// // here's an enode with some meaningless child ids
130    /// let enode = SymbolLang::new("*", vec![Id::from(0), Id::from(0)]);
131    /// // make a new recexpr, replacing enode's children with a_plus_2
132    /// let recexpr = enode.join_recexprs(|_id| &a_plus_2);
133    /// assert_eq!(recexpr, "(* (+ a 2) (+ a 2))".parse().unwrap())
134    /// ```
135    fn join_recexprs<F, Expr>(&self, mut child_recexpr: F) -> RecExpr<Self>
136    where
137        F: FnMut(Id) -> Expr,
138        Expr: AsRef<[Self]>,
139    {
140        fn build<L: Language>(to: &mut RecExpr<L>, from: &[L]) -> Id {
141            let last = from.last().unwrap().clone();
142            let new_node = last.map_children(|id| {
143                let i = usize::from(id) + 1;
144                build(to, &from[0..i])
145            });
146            to.add(new_node)
147        }
148
149        let mut expr = RecExpr::default();
150        let node = self
151            .clone()
152            .map_children(|id| build(&mut expr, child_recexpr(id).as_ref()));
153        expr.add(node);
154        expr
155    }
156
157    /// Build a [`RecExpr`] from an e-node.
158    ///
159    /// The provided `get_node` function must return the same node for a given
160    /// [`Id`] on multiple invocations.
161    ///
162    /// # Example
163    ///
164    /// You could use this method to perform an "ad-hoc" extraction from the e-graph,
165    /// where you already know which node you want pick for each class:
166    /// ```
167    /// # use egg::*;
168    /// let mut egraph = EGraph::<SymbolLang, ()>::default();
169    /// let expr = "(foo (bar1 (bar2 (bar3 baz))))".parse().unwrap();
170    /// let root = egraph.add_expr(&expr);
171    /// let get_first_enode = |id| egraph[id].nodes[0].clone();
172    /// let expr2 = get_first_enode(root).build_recexpr(get_first_enode);
173    /// assert_eq!(expr, expr2)
174    /// ```
175    fn build_recexpr<F>(&self, mut get_node: F) -> RecExpr<Self>
176    where
177        F: FnMut(Id) -> Self,
178    {
179        self.try_build_recexpr::<_, std::convert::Infallible>(|id| Ok(get_node(id)))
180            .unwrap()
181    }
182
183    /// Same as [`Language::build_recexpr`], but fallible.
184    fn try_build_recexpr<F, Err>(&self, mut get_node: F) -> Result<RecExpr<Self>, Err>
185    where
186        F: FnMut(Id) -> Result<Self, Err>,
187    {
188        let mut set = IndexSet::<Self>::default();
189        let mut ids = HashMap::<Id, Id>::default();
190        let mut todo = self.children().to_vec();
191
192        while let Some(id) = todo.last().copied() {
193            if ids.contains_key(&id) {
194                todo.pop();
195                continue;
196            }
197
198            let node = get_node(id)?;
199
200            // check to see if we can do this node yet
201            let mut ids_has_all_children = true;
202            for child in node.children() {
203                if !ids.contains_key(child) {
204                    ids_has_all_children = false;
205                    todo.push(*child)
206                }
207            }
208
209            // all children are processed, so we can lookup this node safely
210            if ids_has_all_children {
211                let node = node.map_children(|id| ids[&id]);
212                let new_id = set.insert_full(node).0;
213                ids.insert(id, Id::from(new_id));
214                todo.pop();
215            }
216        }
217
218        // finally, create the expression and add the root node
219        let mut expr: RecExpr<_> = set.into_iter().collect();
220        expr.add(self.clone().map_children(|id| ids[&id]));
221        Ok(expr)
222    }
223}

225/// A trait for parsing e-nodes. This is implemented automatically by
226/// [`define_language!`].
227///
228/// If a [`Language`] implements both [`Display`] and [`FromOp`], the
229/// [`Display`] implementation should produce a string suitable for parsing by
230/// [`from_op`]:
231///
232/// ```
233/// # use egg::*;
234/// # use std::fmt::Display;
235/// fn from_op_display_compatible<T: FromOp + Display>(node: T) {
236///     let op = node.to_string();
237///     let mut children = Vec::new();
238///     node.for_each(|id| children.push(id));
239///     let parsed = T::from_op(&op, children).unwrap();
240///
241///     assert_eq!(node, parsed);
242/// }
243/// ```
244///
245/// # Examples
246/// `define_language!` implements [`FromOp`] and [`Display`] automatically:
247/// ```
248/// # use egg::*;
249///
250/// define_language! {
251///     enum Calc {
252///        "+" = Add([Id; 2]),
253///        Num(i32),
254///     }
255/// }
256///
257/// let add = Calc::Add([Id::from(0), Id::from(1)]);
258/// let parsed = Calc::from_op("+", vec![Id::from(0), Id::from(1)]).unwrap();
259///
260/// assert_eq!(add.to_string(), "+");
261/// assert_eq!(parsed, add);
262/// ```
263///
264/// [`from_op`]: FromOp::from_op
265pub trait FromOp: Language + Sized {
266    /// The error type returned by [`from_op`] if its arguments do not
267    /// represent a valid e-node.
268    ///
269    /// [`from_op`]: FromOp::from_op
270    type Error: Debug;
271
272    /// Parse an e-node with operator `op` and children `children`.
273    fn from_op(op: &str, children: Vec<Id>) -> Result<Self, Self::Error>;
274}

302pub trait LanguageChildren {
303    /// Checks if there are no children.
304    fn is_empty(&self) -> bool {
305        self.len() == 0
306    }
307    /// Returns the number of children.
308    fn len(&self) -> usize;
309    /// Checks if n is an acceptable number of children for this type.
310    fn can_be_length(n: usize) -> bool;
311    /// Create an instance of this type from a `Vec<Id>`,
312    /// with the guarantee that can_be_length is already true on the `Vec`.

314    /// Returns a slice of the children `Id`s.
315    fn as_slice(&self) -> &[Id];
316    /// Returns a mutable slice of the children `Id`s.
317    fn as_mut_slice(&mut self) -> &mut [Id];
318}

369/// A recursive expression from a user-defined [`Language`].
370///
371/// This conceptually represents a recursive expression, but it's actually just
372/// a list of enodes.
373///
374/// [`RecExpr`]s must satisfy the invariant that enodes' children must refer to
375/// elements that come before it in the list. For example, the expression
376/// `(+ (* x 5) x)` could be represented by a recursive expression of the form
377/// `[Num(5), Var("x"), Mul(1, 0), Add(2, 1)]`.
378///
379/// If the `serde-1` feature is enabled, this implements
380/// [`serde::Serialize`](https://docs.rs/serde/latest/serde/trait.Serialize.html).
381#[derive(Debug, Clone, PartialEq, Eq, PartialOrd, Ord, Hash)]
382pub struct RecExpr<L> {
383    nodes: Vec<L>,
384}

459impl<L: Language> RecExpr<L> {
460    /// Adds a given enode to this `RecExpr`.
461    /// The enode's children [`Id`]s must refer to elements already in this list.
462    pub fn add(&mut self, node: L) -> Id {
463        debug_assert!(
464            node.all(|id| id <= self.root()),
465            "node {:?} has children not in this expr: {:?}",
466            node,
467            self
468        );
469        self.nodes.push(node);
470        self.root()
471    }
472
473    pub(crate) fn compact(mut self) -> Self {
474        let mut ids = hashmap_with_capacity::<Id, Id>(self.len());
475        let mut set = IndexSet::default();
476        for (i, node) in self.nodes.drain(..).enumerate() {
477            let node = node.map_children(|id| ids[&id]);
478            let new_id = set.insert_full(node).0;
479            ids.insert(Id::from(i), Id::from(new_id));
480        }
481        self.nodes.extend(set);
482        self
483    }
484
485    pub(crate) fn extract(&self, new_root: Id) -> Self {
486        self[new_root].build_recexpr(|id| self[id].clone())
487    }
488
489    /// Returns an iterator over the [`Id`]s in this expression.
490    pub fn ids(&self) -> impl ExactSizeIterator<Item = Id> + DoubleEndedIterator {
491        (0..self.len()).map(Id::from)
492    }
493
494    /// Returns an iterator over the [`Id`]s and enodes of this expression.
495    pub fn items(&self) -> impl ExactSizeIterator<Item = (Id, &L)> + DoubleEndedIterator {
496        self.ids().zip(self)
497    }
498
499    /// Returns an iterator over the [`Id`]s and enodes of this expression.
500    pub fn items_mut(
501        &mut self,
502    ) -> impl ExactSizeIterator<Item = (Id, &mut L)> + DoubleEndedIterator {
503        let ids = (0..self.len()).map(Id::from);
504        ids.zip(self.iter_mut())
505    }
506
507    /// Checks if this expr is a DAG, i.e. doesn't have any back edges
508    pub fn is_dag(&self) -> bool {
509        self.items().all(|(id, n)| n.all(|child| child < id))
510    }
511
512    /// Get the root node of this expression. When adding a new node via `add`, it becomes the new root.
513    pub fn root(&self) -> Id {
514        self.ids().last().unwrap()
515    }
516}

569impl<L: Language + Display> RecExpr<L> {
570    /// Convert this RecExpr into an Sexp
571    pub(crate) fn to_sexp(&self) -> Sexp {
572        let last = self.nodes.len() - 1;
573        if !self.is_dag() {
574            log::warn!("Tried to print a non-dag: {:?}", self.nodes);
575        }
576        self.to_sexp_rec(last, &mut |_| None)
577    }
578
579    fn to_sexp_rec(&self, i: usize, f: &mut impl FnMut(usize) -> Option<String>) -> Sexp {
580        let node = &self.nodes[i];
581        let op = Sexp::String(node.to_string());
582        if node.is_leaf() {
583            op
584        } else {
585            let mut vec = vec![op];
586            for child in node.children().iter().map(|i| usize::from(*i)) {
587                vec.push(if let Some(s) = f(child) {
588                    return Sexp::String(s);
589                } else if child < i {
590                    self.to_sexp_rec(child, f)
591                } else {
592                    Sexp::String(format!("<<<< CYCLE to {} = {:?} >>>>", i, node))
593                })
594            }
595            Sexp::List(vec)
596        }
597    }
598
599    /// Pretty print with a maximum line length.
600    ///
601    /// This gives you a nice, indented, pretty-printed s-expression.
602    ///
603    /// # Example
604    /// ```
605    /// # use egg::*;
606    /// let e: RecExpr<SymbolLang> = "(* (+ 2 2) (+ x y))".parse().unwrap();
607    /// assert_eq!(e.pretty(10), "
608    /// (*
609    ///   (+ 2 2)
610    ///   (+ x y))
611    /// ".trim());
612    /// ```
613    pub fn pretty(&self, width: usize) -> String {
614        let sexp = self.to_sexp();
615
616        let mut buf = String::new();
617        pretty_print(&mut buf, &sexp, width, 1).unwrap();
618        buf
619    }
620}

622/// An error type for failures when attempting to parse an s-expression as a
623/// [`RecExpr<L>`].
624#[derive(Debug, Error)]
625pub enum RecExprParseError<E> {
626    /// An empty s-expression was found. Usually this is caused by an
627    /// empty list "()" somewhere in the input.
628    #[error("found empty s-expression")]
629    EmptySexp,
630
631    /// A list was found where an operator was expected. This is caused by
632    /// s-expressions of the form "((a b c) d e f)."
633    #[error("found a list in the head position: {0}")]
634    HeadList(Sexp),
635
636    /// Attempting to parse an operator into a value of type `L` failed.
637    #[error(transparent)]
638    BadOp(E),
639
640    /// An error occurred while parsing the s-expression itself, generally
641    /// because the input had an invalid structure (e.g. unpaired parentheses).
642    #[error(transparent)]
643    BadSexp(SexpError),
644}
645
646impl<L: FromOp> FromStr for RecExpr<L> {
647    type Err = RecExprParseError<L::Error>;
648
649    fn from_str(s: &str) -> Result<Self, Self::Err> {
650        use RecExprParseError::*;
651
652        fn parse_sexp_into<L: FromOp>(
653            sexp: &Sexp,
654            expr: &mut RecExpr<L>,
655        ) -> Result<Id, RecExprParseError<L::Error>> {
656            match sexp {
657                Sexp::Empty => Err(EmptySexp),
658                Sexp::String(s) => {
659                    let node = L::from_op(s, vec![]).map_err(BadOp)?;
660                    Ok(expr.add(node))
661                }
662                Sexp::List(list) if list.is_empty() => Err(EmptySexp),
663                Sexp::List(list) => match &list[0] {
664                    Sexp::Empty => unreachable!("Cannot be in head position"),
665                    list @ Sexp::List(..) => Err(HeadList(list.to_owned())),
666                    Sexp::String(op) => {
667                        let arg_ids: Vec<Id> = list[1..]
668                            .iter()
669                            .map(|s| parse_sexp_into(s, expr))
670                            .collect::<Result<_, _>>()?;
671                        let node = L::from_op(op, arg_ids).map_err(BadOp)?;
672                        Ok(expr.add(node))
673                    }
674                },
675            }
676        }
677
678        let mut expr = RecExpr::default();
679        let sexp = symbolic_expressions::parser::parse_str(s.trim()).map_err(BadSexp)?;
680        parse_sexp_into(&sexp, &mut expr)?;
681        Ok(expr)
682    }
683}

685/// Result of [`Analysis::merge`] indicating which of the inputs
686/// are different from the merged result.
687///
688/// The fields correspond to whether the initial `a` and `b` inputs to [`Analysis::merge`]
689/// were different from the final merged value.
690///
691/// In both cases the result may be conservative -- they may indicate `true` even
692/// when there is no difference between the input and the result.
693///
694/// `DidMerge`s can be "or"ed together using the `|` operator.
695/// This can be useful for composing analyses.
696pub struct DidMerge(pub bool, pub bool);
697
698impl BitOr for DidMerge {
699    type Output = DidMerge;
700
701    fn bitor(mut self, rhs: Self) -> Self::Output {
702        self.0 |= rhs.0;
703        self.1 |= rhs.1;
704        self
705    }
706}

708/** Arbitrary data associated with an [`EClass`].
709
710`egg` allows you to associate arbitrary data with each eclass.
711The [`Analysis`] allows that data to behave well even across eclasses merges.
712
713[`Analysis`] can prove useful in many situtations.
714One common one is constant folding, a kind of partial evaluation.
715In that case, the metadata is basically `Option<L>`, storing
716the cheapest constant expression (if any) that's equivalent to the
717enodes in this eclass.
718See the test files [`math.rs`] and [`prop.rs`] for more complex
719examples on this usage of [`Analysis`].
720
721If you don't care about [`Analysis`], `()` implements it trivally,
722just use that.
723
724# Example
725
726```
727use egg::{*, rewrite as rw};
728
729define_language! {
730    enum SimpleMath {
731        "+" = Add([Id; 2]),
732        "*" = Mul([Id; 2]),
733        Num(i32),
734        Symbol(Symbol),
735    }
736}
737
738// in this case, our analysis itself doesn't require any data, so we can just
739// use a unit struct and derive Default
740#[derive(Default)]
741struct ConstantFolding;
742impl Analysis<SimpleMath> for ConstantFolding {
743    type Data = Option<i32>;
744
745    fn merge(&mut self, to: &mut Self::Data, from: Self::Data) -> DidMerge {
746        egg::merge_max(to, from)
747    }
748
749    fn make(egraph: &mut EGraph<SimpleMath, Self>, enode: &SimpleMath, _id: Id) -> Self::Data {
750        let x = |i: &Id| egraph[*i].data;
751        match enode {
752            SimpleMath::Num(n) => Some(*n),
753            SimpleMath::Add([a, b]) => Some(x(a)? + x(b)?),
754            SimpleMath::Mul([a, b]) => Some(x(a)? * x(b)?),
755            _ => None,
756        }
757    }
758
759    fn modify(egraph: &mut EGraph<SimpleMath, Self>, id: Id) {
760        if let Some(i) = egraph[id].data {
761            let added = egraph.add(SimpleMath::Num(i));
762            egraph.union(id, added);
763        }
764    }
765}
766
767let rules = &[
768    rw!("commute-add"; "(+ ?a ?b)" => "(+ ?b ?a)"),
769    rw!("commute-mul"; "(* ?a ?b)" => "(* ?b ?a)"),
770
771    rw!("add-0"; "(+ ?a 0)" => "?a"),
772    rw!("mul-0"; "(* ?a 0)" => "0"),
773    rw!("mul-1"; "(* ?a 1)" => "?a"),
774];
775
776let expr = "(+ 0 (* (+ 4 -3) foo))".parse().unwrap();
777let mut runner = Runner::<SimpleMath, ConstantFolding, ()>::default().with_expr(&expr).run(rules);
778let just_foo = runner.egraph.add_expr(&"foo".parse().unwrap());
779assert_eq!(runner.egraph.find(runner.roots[0]), runner.egraph.find(just_foo));
780```
781
782[`math.rs`]: https://github.com/egraphs-good/egg/blob/main/tests/math.rs
783[`prop.rs`]: https://github.com/egraphs-good/egg/blob/main/tests/prop.rs
784*/
785pub trait Analysis<L: Language>: Sized {
786    /// The per-[`EClass`] data for this analysis.
787    type Data: Debug;
788
789    /// Makes a new [`Analysis`] data for a given e-node that will go into the given e-class.
790    ///
791    /// Note the mutable `egraph` parameter: this is needed for some
792    /// advanced use cases, but most use cases will not need to mutate
793    /// the e-graph in any way.
794    /// It is **not** `make`'s responsiblity to insert the e-node;
795    /// the e-node is "being inserted" when this function is called.
796    /// Doing so will create an infinite loop.
797    ///
798    /// Note that `enode`'s children may not be canonical
799    fn make(egraph: &mut EGraph<L, Self>, enode: &L, id: Id) -> Self::Data;
800
801    /// Same as [`Analysis::make`], but called during rebuilding.
802    ///
803    /// By default, it just calls `make`.
804    fn remake(egraph: &mut EGraph<L, Self>, enode: &L, id: Id) -> Self::Data {
805        Self::make(egraph, enode, id)
806    }
807
808    /// An optional hook that allows inspection before a [`union`] occurs.
809    /// When explanations are enabled, it gives two ids that represent the two particular terms being unioned, not the canonical ids for the two eclasses.
810    /// It also gives a justification for the union when explanations are enabled.
811    ///
812    /// By default it does nothing.
813    ///
814    /// `pre_union` is called _a lot_, so doing anything significant
815    /// (like printing) will cause things to slow down.
816    ///
817    /// [`union`]: EGraph::union()
818    #[allow(unused_variables)]
819    fn pre_union(
820        egraph: &EGraph<L, Self>,
821        id1: Id,
822        id2: Id,
823        justification: &Option<Justification>,
824    ) {
825    }
826
827    /// Defines how to merge two `Data`s when their containing
828    /// [`EClass`]es merge.
829    ///
830    /// This should update `a` to correspond to the merged analysis
831    /// data.
832    ///
833    /// The result is a `DidMerge(a_merged, b_merged)` indicating whether
834    /// the merged result is different from `a` and `b` respectively.
835    ///
836    /// Since `merge` can modify `a`, let `a0`/`a1` be the value of `a`
837    /// before/after the call to `merge`, respectively.
838    ///
839    /// If `a0 != a1` the result must have `a_merged == true`. This may be
840    /// conservative -- it may be `true` even if even if `a0 == a1`.
841    ///
842    /// If `b != a1` the result must have `b_merged == true`. This may be
843    /// conservative -- it may be `true` even if even if `b == a1`.
844    ///
845    /// This function may modify the [`Analysis`], which can be useful as a way
846    /// to store information for the [`Analysis::modify`] hook to process, since
847    /// `modify` has access to the e-graph.
848    fn merge(&mut self, a: &mut Self::Data, b: Self::Data) -> DidMerge;
849
850    /// A hook that allows the modification of the
851    /// [`EGraph`].
852    ///
853    /// By default this does nothing.
854    ///
855    /// This function is called immediately following
856    /// `Analysis::merge` when unions are performed.
857    #[allow(unused_variables)]
858    fn modify(egraph: &mut EGraph<L, Self>, id: Id) {}
859
860    /// Whether or not e-matching should allow finding cycles.
861    ///
862    /// By default, this returns `true`.
863    ///
864    /// Setting this to `false` can improve performance in some cases, but risks
865    /// missing some equalities depending on the use case.
866    fn allow_ematching_cycles(&self) -> bool {
867        true
868    }
869}
870
871impl<L: Language> Analysis<L> for () {
872    type Data = ();
873    fn make(_egraph: &mut EGraph<L, Self>, _enode: &L, _id: Id) -> Self::Data {}
874    fn merge(&mut self, _: &mut Self::Data, _: Self::Data) -> DidMerge {
875        DidMerge(false, false)
876    }
877}

879/// A utility for implementing [`Analysis::merge`]
880/// when the `Data` type has a total ordering.
881/// This will take the maximum of the two values.
882pub fn merge_max<T: Ord>(to: &mut T, from: T) -> DidMerge {
883    let cmp = (*to).cmp(&from);
884    match cmp {
885        Ordering::Less => {
886            *to = from;
887            DidMerge(true, false)
888        }
889        Ordering::Equal => DidMerge(false, false),
890        Ordering::Greater => DidMerge(false, true),
891    }
892}
893
894/// A utility for implementing [`Analysis::merge`]
895/// when the `Data` type has a total ordering.
896/// This will take the minimum of the two values.
897pub fn merge_min<T: Ord>(to: &mut T, from: T) -> DidMerge {
898    let cmp = (*to).cmp(&from);
899    match cmp {
900        Ordering::Less => DidMerge(false, true),
901        Ordering::Equal => DidMerge(false, false),
902        Ordering::Greater => {
903            *to = from;
904            DidMerge(true, false)
905        }
906    }
907}
908
909/// A utility for implementing [`Analysis::merge`]
910/// when the `Data` type is an [`Option`].
911///
912/// Always take a `Some` over a `None`
913/// and calls the given function to merge two `Some`s.
914pub fn merge_option<T>(
915    to: &mut Option<T>,
916    from: Option<T>,
917    merge_fn: impl FnOnce(&mut T, T) -> DidMerge,
918) -> DidMerge {
919    match (to.as_mut(), from) {
920        (None, None) => DidMerge(false, false),
921        (None, from @ Some(_)) => {
922            *to = from;
923            DidMerge(true, false)
924        }
925        (Some(_), None) => DidMerge(false, true),
926        (Some(a), Some(b)) => merge_fn(a, b),
927    }
928}
